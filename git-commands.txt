# Git Commands to Run After Installing Git
# Copy and paste these commands one by one into your terminal

# 1. Initialize the repository
git init

# 2. Add all files
git add .

# 3. Create initial commit
git commit -m "Initial commit - Scronth social media platform"

# 4. Set main branch
git branch -M main

# 5. Add remote repository
git remote add origin https://github.com/silaslukep/Scronth.git

# 6. Push to GitHub (you'll need to authenticate)
git push -u origin main
